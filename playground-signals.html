Phoenix LiveView — Catalog Sidebar Implementation Spec

Module Structure

  • IchorWeb.SignalsLive — root LiveView; owns feed state, filter assigns, and sidebar assigns
  • IchorWeb.Components.SignalCatalog — stateless functional component; renders accordion catalog with search, rates, and pulse
  • IchorWeb.Components.SignalEmitter — stateful live component; handles emit form state and result display
  • IchorWeb.Components.SignalFeedRow — functional component; renders one feed row with expansion
  • IchorWeb.Hooks.CatalogSearch — JS hook for client-side catalog filtering (avoids server round-trips)

@active_cat_pulse atom | nil          # category that just fired (CSS animation)

handle_event/3 — Catalog + Emitter

  • "toggle_category_group" — toggle cat atom in @open_categories MapSet
  • "catalog_search" — update @catalog_search; JS hook handles DOM visibility client-side
  • "filter_signal" — set or clear @filters.signal_name (toggle on same)

Rate Calculation + Pulse

def handle_info(:recalc_rates, socket) do
  cutoff = DateTime.add(DateTime.utc_now(), -60, :second)
  rates =
    socket.assigns.signals
    |> Enum.filter(&(DateTime.compare(&1.ts, cutoff) == :gt))
    |> Enum.group_by(& &1.category)
    |> Map.new(fn {cat, sigs} -> {cat, length(sigs)} end)
  Process.send_after(self(), :recalc_rates, 10_000)
  {:noreply, assign(socket, category_rates: rates)}
end

def handle_info({:signal, name, data}, socket) do
  cat = Ichor.Signals.Catalog.lookup_or_derive(name).category
  Process.send_after(self(), {:clear_pulse, cat}, 700)
  # ... ingest signal ...
  {:noreply, assign(socket, active_cat_pulse: cat)}
end

JS Hook: CatalogSearch

// assets/js/hooks/catalog_search.js
const CatalogSearch = {
  mounted() {
    this.el.addEventListener("input", e => {
      const q = e.target.value.toLowerCase();
      document.querySelectorAll("section.category-group").forEach(group => {
        let anyVisible = false;
        group.querySelectorAll("article.signal-item").forEach(item => {
          const name = item.querySelector(".signal-name")?.textContent.toLowerCase() ?? "";
          const doc  = item.querySelector(".signal-doc")?.textContent.toLowerCase()  ?? "";
          const keys = item.querySelector(".signal-keys")?.textContent.toLowerCase() ?? "";
          const show = !q || name.includes(q) || doc.includes(q) || keys.includes(q);
          item.style.display = show ? "" : "none";
          if (show) anyVisible = true;
        });
        group.style.display = anyVisible ? "" : "none";
        if (q && anyVisible) group.classList.add("open");
      });
    });
  }
};
export default CatalogSearch;
